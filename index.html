const SHEET_ID = '1Btk5yz7D2_0HVew1iyediOwpwTIFbvJVQfcK7ENEg3A'; const PERMIT_FOLDER_ID = '1CRJCk37ycQlAw3ZgjOkpUUNAl9tioM9M'; function doGet(e) { if (!e.parameter.action) { return HtmlService.createHtmlOutput("API is running. Access via GitHub Dashboard."); } try { let result; const action = e.parameter.action; if (action === 'getEvents') result = getSheetDataAsObjects('EVENTS'); else if (action === 'getPayments') result = getSheetDataAsObjects('PAYMENTS'); return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({error: err.toString()})).setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { const data = JSON.parse(e.postData.contents); const action = data.action; let result; if (action === 'addNewEvent') { result = addNewEvent(data.formObject); } else if (action === 'uploadPermit') { result = uploadPermitFile(data.base64, data.filename, data.eventId); } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({success: false, error: "Backend Error: " + err.toString()})).setMimeType(ContentService.MimeType.JSON); } } function getSheetDataAsObjects(sheetName) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(sheetName); if (!sheet) return []; const data = sheet.getDataRange().getDisplayValues(); if (data.length <= 1) return []; const headers = data.shift(); return data.map(row => { let obj = {}; headers.forEach((header, i) => obj[header.trim()] = row[i]); return obj; }); } function addNewEvent(formObject) { const eventSheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName('EVENTS'); const paymentSheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName('PAYMENTS'); const nextNum = calculateNextIdNumber(eventSheet); const newId = 'egm' + ('0000' + nextNum).slice(-4); let servicesStr = Array.isArray(formObject.services) ? formObject.services.join(", ") : (formObject.services || ""); const eventRow = [newId, formObject.eventDate || "", formObject.eventName || "", formObject.eventStatus || "Pencil Booked", formObject.avp ? "TRUE" : "FALSE", servicesStr, formObject.startTime || "", formObject.callTime || "", formObject.manpower || "", formObject.contactPerson || "", formObject.contactNumber || "", formObject.notes || "", ""]; eventSheet.appendRow(eventRow); if (formObject.paymentType) { const paymentRow = [newId, formObject.paymentType, formObject.emailThread || "", formObject.billingStatus || "", formObject.billingRef || ""]; paymentSheet.appendRow(paymentRow); } return { success: true, id: newId }; } function calculateNextIdNumber(sheet) { const data = sheet.getRange("A2:A").getValues(); let maxId = 0; data.forEach(row => { const id = row[0].toString().toLowerCase(); if (id.startsWith('egm')) { const num = parseInt(id.replace('egm', ''), 10); if (!isNaN(num) && num > maxId) maxId = num; } }); return maxId + 1; } function uploadPermitFile(base64Data, filename, eventId) { try { const folder = DriveApp.getFolderById(PERMIT_FOLDER_ID.trim()); const contentType = base64Data.substring(5, base64Data.indexOf(';')); const bytes = Utilities.base64Decode(base64Data.substr(base64Data.indexOf('base64,') + 7)); // Naming: egm####_filename const blob = Utilities.newBlob(bytes, contentType, eventId + "_" + filename); const file = folder.createFile(blob); const fileUrl = file.getUrl(); const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName('EVENTS'); const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const permitCol = headers.indexOf('Permits') + 1; const data = sheet.getDataRange().getValues(); for (let i = 0; i < data.length; i++) { if (data[i][0] == eventId) { sheet.getRange(i + 1, permitCol).setValue(fileUrl); break; } } return { success: true, url: fileUrl }; } catch (e) { return { success: false, error: e.toString() }; } }
